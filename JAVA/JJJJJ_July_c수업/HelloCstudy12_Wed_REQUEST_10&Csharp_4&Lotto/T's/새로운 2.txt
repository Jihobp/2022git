int clock(int sec) //int 는 반환하겠다는 값이 정수란것~~암시.
{	
	if(sec>3600)
	{
		printf("1시간 이상\n");
		return -1;  //Q. 임의???????????????????????????????????????????
	}
	printf("%d분 %d초\n", sec / 60, sec % 60);
	return sec / 60; //반환한다 123>2 반환한다  %나머지는 초 //123은 2분,
	
	//return 은 저장한다, 함수에 저장한다.
	//함수으 ㅣ리턴 값은 함수문 안에서 저장된다.
	//메인의 리턴 값은 뭐가 됐던 중요치 않은데 결국 0 값이 종료 및 실행이 잘 된다는 의미이다?
}
